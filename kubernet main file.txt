apiVersion: v1
kind: Pod
metadata:
  name: frontend-backend-pod
  labels:
    app: frontend-backend

  containers:
  - name: angular-frontend
    image: rodeaayush240/angular:frontend
    ports:
    - containerPort: 80
      protocol: TCP

    image: rodeaayush240/angular:backend
